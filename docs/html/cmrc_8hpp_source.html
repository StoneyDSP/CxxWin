308inline std::pair<std::string, std::string> SplitPath(const std::string& path)
309{
310 auto first_sep = path.find("/");
311
312 if (first_sep == path.npos)
313 {
314 return std::make_pair(path, "");

316 return std::make_pair(path.substr(0, first_sep), path.substr(first_sep + 1));

390 const FileOrDirectory* Get(const std::string& path) const
391 {
392 auto pair = SplitPath(path);
393 auto child = _index.find(pair.first);
394
395 if (child == _index.end())
396 {
397 return nullptr;
398 }
399 auto& entry = child->second;
400
401 if (pair.second.empty())
402 {
403 // We're at the end of the path
404 return &entry;
405 }
406
407 if (entry.IsFile())
408 {
409 // We can't traverse into a file. Stop.
410 return nullptr;
411 }
412
413 // Keep going down
414 return entry.AsDirectory().Get(pair.second);

551inline std::string NormalizePath(std::string path)
552{
553 while (path.find("/") == 0)
554 {
555 path.erase(path.begin());
556 }
557
558 while (!path.empty() && (path.rfind("/") == path.size() - 1))
559 {
560 path.pop_back();
561 }
562
563 auto off = path.npos;
564
565 while ((off = path.find("//")) != path.npos)
566 {
567 path.erase(path.begin() + static_cast<std::string::difference_type>(off));
568 }
569
570 return path;
571}

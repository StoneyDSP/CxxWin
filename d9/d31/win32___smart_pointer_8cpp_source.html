354 std::make_tuple (__uuidof(ComClasses), static_cast<void*> (static_cast<ComClasses*> (this)))...
355 };
356
357 // Query a ranged 'for' loop (each 'base' in the 'bases' tuple array)
358 for (CONST auto& base : bases)
